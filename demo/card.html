# 使用方法

# 实现方法

<template>
  <div class="card">
    <div class="title">{{title}}</div>
    <div class="desc">{{desc}}</div>
  </div>
</template>

<script>
  export default {
    name: "card-me",
    props:['title','desc']
  }
</script>

<style scoped lang="scss">

      }
